<!-- New CompositeSurrogate concept -->
```rust
pub struct CompositeSurrogate {
    components: Vec<ComponentSurrogate>,
}
impl CompositeSurrogate {
    pub fn new(components: Vec<ComponentSurrogate>) -> Self;
    pub fn predict_loads(&self, current_temps: &[f64]) -> Vec<f64> {
        // Sum predictions from each component
        let mut sum = vec![0.0; current_temps.len()];
        for comp in &self.components {
            let pred = comp.predict_loads(current_temps)?;
            for (s, &p) in sum.iter_mut().zip(pred.iter()) { *s += p; }
        }
        sum
    }
}
pub struct ComponentSurrogate {
    name: String,
    manager: SurrogateManager, // each component may have its own ONNX model
}
```
