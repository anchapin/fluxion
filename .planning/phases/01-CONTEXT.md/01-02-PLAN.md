impl Construction {
    /// Calculates exterior-to-mass conductance (h_tr_em) in W/K
    /// Combines wall/roof/floor conductances with window U-value effects
    /// Accounts for thermal bridge effects at edges and corners
    ///
    /// Formula: h_tr_em = (U_wall * A_wall + U_roof * A_roof + U_floor * A_floor + U_window * A_window) / area_ratio
    /// where area_ratio accounts for thermal mass surface area vs exterior surface area
    pub fn calc_h_tr_em(&self, window_u_value: f64, surface_area: f64) -> f64 {
        // Calculate total thermal conductance from exterior to thermal mass
        // Include wall, roof, floor, and window contributions
        let wall_u = 1.0 / self.wall.total_r_value();
        let roof_u = 1.0 / self.roof.total_r_value();
        let floor_u = 1.0 / self.floor.total_r_value();

        // Exterior surface areas (from geometry, assume 1:1 for single zone)
        let wall_area = surface_area * 0.6; // Approximate wall-to-floor ratio
        let roof_area = surface_area * 1.0; // Roof area = floor area
        let floor_area = surface_area * 1.0; // Floor area = floor area
        let window_area = wall_area * 0.25; // Window-to-wall ratio (ASHRAE 140 typical)

        // Total conductance (W/K)
        let total_conductance = (wall_u * wall_area) + (roof_u * roof_area) + (floor_u * floor_area) + (window_u_value * window_area);

        // Apply thermal bridge factor (1.0 for simple cases, >1.0 for complex geometry)
        let thermal_bridge_factor = 1.1; // 10% enhancement for thermal bridges

        total_conductance * thermal_bridge_factor
    }

    /// Calculates window conductance (h_tr_w) in W/K
    /// Exterior-to-interior through glazed surfaces only
    ///
    /// Formula: h_tr_w = U_window * A_window
    pub fn calc_h_tr_w(&self, window_u_value: f64, window_area: f64) -> f64 {
        window_u_value * window_area
    }

    /// Calculates mass-to-surface conductance (h_tr_ms) in W/K
    /// Thermal coupling between thermal mass and interior surface
    ///
    /// Formula: h_tr_ms = (Cm / (ρ * cp * V)) * (A_mass / A_surface)
    /// where Cm is thermal capacitance, ρ is density, cp is specific heat, V is volume
    pub fn calc_h_tr_ms(&self, surface_area: f64) -> f64 {
        // Simplified: proportional to surface area with material properties
        // For lightweight construction (ASHRAE 140 Case 600):
        let material_conductivity = 0.04; // W/K per m² (typical for lightweight walls)
        material_conductivity * surface_area
    }

    /// Calculates surface-to-interior conductance (h_tr_is) in W/K
    /// Interior convective/radiative heat transfer from surface to zone air
    ///
    /// Formula: h_tr_is = h_conv * A_surface + h_rad * A_surface
    /// where h_conv is convective coefficient (3-8 W/m²K), h_rad is radiative (4-6 W/m²K)
    pub fn calc_h_tr_is(&self, surface_area: f64) -> f64 {
        // ASHRAE film coefficients: convective ~3 W/m²K, radiative ~5 W/m²K
        let h_conv = 3.0; // W/m²K
        let h_rad = 5.0; // W/m²K
        let combined_coefficient = h_conv + h_rad; // 8 W/m²K

        combined_coefficient * surface_area
    }
}

impl Assemblies {
    /// Calculates ventilation conductance (h_ve) in W/K from air change rate
    ///
    /// Formula: h_ve = ρ_air * cp_air * ACH * V_zone / 3600
    /// where ρ_air = 1.2 kg/m³, cp_air = 1005 J/kgK, ACH is air changes per hour, V_zone is zone volume
    pub fn calc_h_ve(&self, ach: f64, zone_volume: f64) -> f64 {
        const AIR_DENSITY: f64 = 1.2; // kg/m³
        const AIR_SPECIFIC_HEAT: f64 = 1005.0; // J/kgK
        const SECONDS_PER_HOUR: f64 = 3600.0;

        // Mass flow rate (kg/s) = ρ * V * ACH / 3600
        let mass_flow_rate = AIR_DENSITY * zone_volume * ach / SECONDS_PER_HOUR;

        // Conductance (W/K) = mass_flow_rate * cp
        mass_flow_rate * AIR_SPECIFIC_HEAT
    }
}
