```rust
impl ThermalModel {
    /// Calculates free-floating zone temperature (Ti_free)
    /// This is the temperature the zone would reach without HVAC control
    /// Based on energy balance of 5R1C thermal network
    ///
    /// Formula from ISO 13790:
    /// Ti_free = (Φ_ist + Φ_sol + Φ_int) / H_tr
    /// where H_tr = h_tr_em + h_tr_w + h_ve (total heat loss coefficient)
    /// Φ_ist = heat from internal mass, Φ_sol = solar gains, Φ_int = internal gains
    fn calculate_ti_free(&self, hour: usize) -> f64 {
        // Total heat loss coefficient (W/K)
        let h_tr = self.h_tr_em.get(hour) + self.h_tr_w.get(hour) + self.h_ve.get(hour);

        // Heat sources (W)
        let phi_ist = self.mass_temperatures.get(hour) * self.h_tr_ms.get(hour);
        let phi_sol = self.loads.get(hour); // Solar gains (simplified)
        let phi_int = 200.0; // Internal gains (200W typical for ASHRAE 140)

        // External temperature (simplified, use weather data in production)
        let t_ext = 10.0; // 10°C typical for Denver winter

        // Free-floating temperature (°C)
        let ti_free = (phi_ist + phi_sol + phi_int) / h_tr + t_ext;

        ti_free
    }
}
```

2. **Update IdealHVACController to use Ti_free:**

```rust
impl IdealHVACController {
    /// Calculates HVAC load required to maintain setpoint
    /// Uses Ti_free (free-floating temperature) not Ti (actual zone temperature)
    ///
    /// Returns positive value for heating, negative for cooling
    pub fn calculate_load(&self, ti_free: f64, ti_actual: f64) -> f64 {
        let mode = self.get_hvac_mode(ti_actual);

        match mode {
            HVACMode::Heating => {
                // Heating needed: Ti_free < heating_setpoint
                // Load = H_tr * (heating_setpoint - Ti_free)
                let delta_t = self.heating_setpoint - ti_free;
                // Use total heat loss coefficient (simplified)
                let h_tr = 100.0; // W/K (typical for ASHRAE 140 Case 600)
                h_tr * delta_t
            }
            HVACMode::Cooling => {
                // Cooling needed: Ti_free > cooling_setpoint
                // Load = H_tr * (cooling_setpoint - Ti_free) [negative value]
                let delta_t = self.cooling_setpoint - ti_free;
                let h_tr = 100.0; // W/K (typical for ASHRAE 140 Case 600)
                h_tr * delta_t // Negative value indicates cooling
            }
            HVACMode::Off => 0.0,
        }
    }

    /// Determines HVAC mode based on zone temperature vs setpoints
    pub fn get_hvac_mode(&self, ti: f64) -> HVACMode {
        if ti < self.heating_setpoint - self.deadband_tolerance {
            HVACMode::Heating
        } else if ti > self.cooling_setpoint + self.deadband_tolerance {
            HVACMode::Cooling
        } else {
            HVACMode::Off
        }
    }
}
```

```rust
impl ThermalModel {
    pub fn solve_timesteps(&mut self, steps: usize, _surrogates: bool, _use_ai: bool) -> f64 {
        let mut total_energy = 0.0;

        for step in 0..steps {
            // Calculate free-floating temperature
            let ti_free = self.calculate_ti_free(step);
            let ti = self.temperatures.get(step);

            // Determine HVAC mode and load using Ti_free
            let hvac_mode = self.hvac_controller.get_hvac_mode(ti);
            let hvac_load = self.hvac_controller.calculate_load(ti_free, ti);

            // Accumulate energy (convert W to kWh: W * 1h / 1000)

    }
```
