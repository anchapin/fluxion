    let t_sunspace_k = self.temperatures[1] + 273.15;
    let t_back_k = self.temperatures[0] + 273.15;
    t_sunspace_k.powi(4) - t_back_k.powi(4)
};

let q_rad = 5.670374419e-8  // Stefan-Boltzmann constant
    * self.surface_emissivity[0]  // ε_back-zone

    * 1.0  // View factor (aligned windows)
    * self.common_wall_area  // Area of common wall (21.6 m² for Case 960)
    * delta_t4_kelvin;

// Ventilation heat transfer (temperature-dependent ACH via stack effect)
let ach_iz = calculate_stack_effect_ach(
    self.temperatures[0],  // T_back-zone
    self.temperatures[1],  // T_sunspace
    self.door_geometry.height,
    self.door_geometry.area,
);
// Use back-zone volume for ventilation calculation
let q_vent = calculate_ventilation_heat_transfer(
    ach_iz,
    self.temperatures[1],  // Source: sunspace (warm in summer, cold in winter)
    self.temperatures[0],  // Target: back-zone

);

// Total inter-zone heat transfer (positive = sunspace → back-zone)
let q_iz_total = q_cond + q_rad + q_vent;
