       fn test_ground_reflected_irradiance() {

           let (_sun_pos, irradiance, _gain) = calculate_hourly_solar(
               39.7392, -104.9903, 2023, 7, 21, 12.0,
               900.0, 150.0, &window, None, None, &[], Orientation::South, Some(0.2)
           );

           // Ground-reflected = GHI * albedo * (1-cos(tilt))/2
           // For vertical surface (tilt=90°), (1-cos(90))/2 = (1-0)/2 = 0.5
           // GHI = beam * cos(incidence) + diffuse (approx)
